// --- DATA FETCHING & SAVING --- window.refreshData = async () => { if(!window.auth.currentUser) return; document.getElementById('loading-indicator').style.display = 'block'; try { const response = await fetch(GOOGLE_URL); const data = await response.json(); state.logs = data.filter(row => row.user === window.auth.currentUser.email); window.renderUI(); } catch (error) { console.error("Data fetch error", error); } document.getElementById('loading-indicator').style.display = 'none'; }; window.deleteLog = async (timestamp) => { if(!confirm("Remove this entry?")) return; document.getElementById('loading-indicator').style.display = 'block'; const payload = { action: 'delete', timestamp: timestamp, user: window.auth.currentUser.email }; try { await fetch(GOOGLE_URL, { method: 'POST', mode: 'no-cors', body: JSON.stringify(payload) }); state.logs = state.logs.filter(l => l.timestamp !== timestamp); window.renderUI(); } catch (e) { alert("Failed to delete"); } document.getElementById('loading-indicator').style.display = 'none'; }; const pushToSheet = async (payload, btnId, defaultText) => { const btn = document.getElementById(btnId); btn.innerText = "Saving..."; try { await fetch(GOOGLE_URL, { method: 'POST', mode: 'no-cors', body: JSON.stringify(payload) }); btn.innerText = "Saved!"; state.logs.push(payload); setTimeout(() => { btn.innerText = defaultText; window.renderUI(); }, 1000); } catch(e) { btn.innerText = "Error!"; setTimeout(() => btn.innerText = defaultText, 2000); } }; window.saveData = (category) => { const amtInput = document.getElementById(category === 'Milk' ? 'milkAmt' : 'formulaAmt'); let val = parseFloat(amtInput.value); if(!val) return alert("Please enter an amount"); if(state.units === 'US') val = val * 29.5735; let detail = category === 'Formula' ? 'Bottle' : `${state.milkMode} (${state.milkSide})`; const timeId = category === 'Milk' ? 'milkTime' : 'formulaTime'; const payload = { action: 'save', timestamp: new Date(document.getElementById(timeId).value).toISOString(), category: category, value: Math.round(val), detail: detail, user: window.auth.currentUser.email }; amtInput.value = ''; pushToSheet(payload, `btn-save-${category.toLowerCase()}`, `Save ${category} Log`); }; window.saveSkin = () => { const start = new Date(document.getElementById('skinStart').value); const end = new Date(document.getElementById('skinEnd').value); const mins = Math.round((end - start) / 60000); if(mins <= 0) return alert("End time must be after start time"); // Changed 'detail' to simply log 'Session' for all caregivers const payload = { action: 'save', timestamp: start.toISOString(), category: 'SkinToSkin', value: mins, detail: 'Session', user: window.auth.currentUser.email }; pushToSheet(payload, 'btn-save-skin', 'Save Session'); }; window.saveNappy = () => { const time = new Date(document.getElementById('nappyTime').value).toISOString(); const payload = { action: 'save', timestamp: time, category: 'Nappy', value: 1, detail: state.nappyType, user: window.auth.currentUser.email }; pushToSheet(payload, 'btn-save-nappy', 'Log Nappy'); }; // --- RENDER LOGIC --- window.renderUI = () => { if(state.tab === 'milk') renderMilk(); if(state.tab === 'formula') renderFormula(); if(state.tab === 'combined') renderCombined(); if(state.tab === 'skin') renderSkin(); if(state.tab === 'nappy') renderNappy(); }; const displayVal = (ml) => state.units === 'US' ? (ml * 0.033814).toFixed(1) : Math.round(ml); const getUnitStr = () => state.units === 'US' ? 'oz' : 'ml'; const getLast7Days = () => { const dates = []; for(let i=6; i>=0; i--) { const d = new Date(); d.setDate(d.getDate() - i); dates.push(d.toLocaleDateString('en-GB', {day: 'numeric', month: 'short'})); } return dates; }; const buildHistoryHTML = (logs, unitStr = '') => { return logs.sort((a,b) => new Date(b.timestamp) - new Date(a.timestamp)).slice(0, 5).map(l => { const time = new Date(l.timestamp).toLocaleString('en-GB', {weekday: 'short', hour: '2-digit', minute:'2-digit'}); // Bulletproof detail reading (filters out literal 'undefined' strings from old bugged data) let logDetail = String(l.detail || l.details || '').replace('undefined', '').trim(); let valStr = l.category === 'SkinToSkin' ? `${l.value} mins` : l.category === 'Nappy' ? l.detail : `${displayVal(l.value)}${unitStr}`; let detailStr = (l.category === 'Nappy' || logDetail === '') ? '' : ` | ${logDetail}`; return `
${valStr} ${detailStr}
${time}
`; }).join(''); }; const updateStats = (id, todayVal, avgVal, unit) => { document.getElementById(`stats-${id}`).innerHTML = `
${todayVal}${unit}
Today
${avgVal}${unit}
7-Day Avg
`; }; const drawChart = (canvasId, datasets, labels) => { const ctx = document.getElementById(canvasId).getContext('2d'); if(charts[canvasId]) charts[canvasId].destroy(); const isNight = document.body.classList.contains('night'); const gridColor = isNight ? 'rgba(255,255,255,0.1)' : 'rgba(0,0,0,0.05)'; const textColor = isNight ? '#ff4d4d' : '#2f4f2f'; charts[canvasId] = new Chart(ctx, { type: 'line', data: { labels: labels, datasets: datasets }, options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { labels: { color: textColor } } }, scales: { x: { grid: { color: gridColor }, ticks: { color: textColor } }, y: { beginAtZero: true, grid: { color: gridColor }, ticks: { color: textColor } } } } }); }; function renderMilk() { const logs = state.logs.filter(l => l.category === 'Milk'); const days = getLast7Days(); const pumpData = days.map(() => 0); const feedData = days.map(() => 0); logs.forEach(l => { const dateStr = new Date(l.timestamp).toLocaleDateString('en-GB', {day: 'numeric', month: 'short'}); const idx = days.indexOf(dateStr); if(idx > -1) { // Bulletproof check to avoid crashing if detail is missing from sheet const dStr = String(l.detail || l.details || '').toLowerCase(); if(dStr.includes('pump')) { pumpData[idx] += parseFloat(l.value || 0); } else { // Defaults to feed if not specifically labelled 'pump' so old data still plots feedData[idx] += parseFloat(l.value || 0); } } }); const todayPump = pumpData[6]; const todayFeed = feedData[6]; const todayTotal = todayPump + todayFeed; const sum7 = pumpData.reduce((a,b)=>a+b,0) + feedData.reduce((a,b)=>a+b,0); const avg = sum7 / 7; updateStats('milk', displayVal(todayTotal), displayVal(avg), getUnitStr()); document.getElementById('history-milk').innerHTML = buildHistoryHTML(logs, getUnitStr()); drawChart('chart-milk', [ { label: 'Pumped', data: pumpData.map(displayVal), borderColor: '#8fbca0', backgroundColor: 'rgba(143,188,160,0.2)', fill: true, tension: 0.4 }, { label: 'Fed', data: feedData.map(displayVal), borderColor: '#d4a373', backgroundColor: 'rgba(212,163,115,0.2)', fill: true, tension: 0.4 } ], days); } function renderFormula() { const logs = state.logs.filter(l => l.category === 'Formula'); const days = getLast7Days(); const data = days.map(() => 0); logs.forEach(l => { const dateStr = new Date(l.timestamp).toLocaleDateString('en-GB', {day: 'numeric', month: 'short'}); const idx = days.indexOf(dateStr); if(idx > -1) data[idx] += parseFloat(l.value || 0); }); const todayTotal = data[6]; const avg = data.reduce((a,b)=>a+b,0) / 7; updateStats('formula', displayVal(todayTotal), displayVal(avg), getUnitStr()); document.getElementById('history-formula').innerHTML = buildHistoryHTML(logs, getUnitStr()); drawChart('chart-formula', [{ label: 'Formula', data: data.map(displayVal), borderColor: '#6a9c7d', backgroundColor: 'rgba(106,156,125,0.2)', fill: true, tension: 0.4 }], days); } function renderCombined() { // Only include milk logs that aren't pumping const milkLogs = state.logs.filter(l => { if (l.category !== 'Milk') return false; const dStr = String(l.detail || l.details || '').toLowerCase(); return !dStr.includes('pump'); }); const formulaLogs = state.logs.filter(l => l.category === 'Formula'); const days = getLast7Days(); const milkData = days.map(() => 0); const formulaData = days.map(() => 0); milkLogs.forEach(l => { const idx = days.indexOf(new Date(l.timestamp).toLocaleDateString('en-GB', {day: 'numeric', month: 'short'})); if(idx > -1) milkData[idx] += parseFloat(l.value || 0); }); formulaLogs.forEach(l => { const idx = days.indexOf(new Date(l.timestamp).toLocaleDateString('en-GB', {day: 'numeric', month: 'short'})); if(idx > -1) formulaData[idx] += parseFloat(l.value || 0); }); const totalData = days.map((_, i) => milkData[i] + formulaData[i]); const todayTotal = totalData[6]; const avg = totalData.reduce((a,b)=>a+b,0) / 7; updateStats('combined', displayVal(todayTotal), displayVal(avg), getUnitStr()); drawChart('chart-combined', [ { label: 'Total Intake', data: totalData.map(displayVal), borderColor: '#2f4f2f', tension: 0.4, borderWidth: 3 }, { label: 'Milk Fed', data: milkData.map(displayVal), borderColor: '#d4a373', tension: 0.4, borderDash: [5, 5] }, { label: 'Formula', data: formulaData.map(displayVal), borderColor: '#6a9c7d', tension: 0.4, borderDash: [5, 5] } ], days); } function renderSkin() { const logs = state.logs.filter(l => l.category === 'SkinToSkin'); const days = getLast7Days(); const data = days.map(() => 0); logs.forEach(l => { const idx = days.indexOf(new Date(l.timestamp).toLocaleDateString('en-GB', {day: 'numeric', month: 'short'})); if(idx > -1) data[idx] += parseFloat(l.value || 0); }); updateStats('skin', data[6], Math.round(data.reduce((a,b)=>a+b,0)/7), 'm'); document.getElementById('history-skin').innerHTML = buildHistoryHTML(logs, ''); drawChart('chart-skin', [{ label: 'Skin to Skin (Mins)', data: data, borderColor: '#e29578', backgroundColor: 'rgba(226,149,120,0.2)', fill: true, tension: 0.4 }], days); } function renderNappy() { const logs = state.logs.filter(l => l.category === 'Nappy'); const days = getLast7Days(); const data = days.map(() => 0); logs.forEach(l => { const idx = days.indexOf(new Date(l.timestamp).toLocaleDateString('en-GB', {day: 'numeric', month: 'short'})); if(idx > -1) data[idx] += 1; }); updateStats('nappy', data[6], (data.reduce((a,b)=>a+b,0)/7).toFixed(1), ''); document.getElementById('history-nappy').innerHTML = buildHistoryHTML(logs, ''); drawChart('chart-nappy', [{ label: 'Total Changes', data: data, borderColor: '#006d77', backgroundColor: 'rgba(0,109,119,0.2)', fill: true, tension: 0.4 }], days); }
